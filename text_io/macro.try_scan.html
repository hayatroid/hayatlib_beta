use text_io::try_scan;

fn parser() -> Result<i32, Box<dyn std::error::Error>> {
    let i: i32;
    let text = "The answer is 42!";

    try_scan!(text.bytes() => "The answer is {}!", i);

    assert_eq!(i, 1);
    Ok(i)
}
